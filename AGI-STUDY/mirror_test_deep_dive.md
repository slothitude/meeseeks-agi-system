                    "sum": lower + upper,
                    "mirror_root": 6 * n
                })

    def _is_twin_prime(self, p1: int, p2: int) -> bool:
        """Check if two numbers are twin primes"""
        return self._is_prime(p1) and self._is_prime(p2)
    
    def _is_prime(self, n: int) -> bool:
        """Simple primality test"""
        if n < 2:
            return False
        for i in range(2, int(n ** 0.5) + 1):
            if n % i == 0:
                return False
        return True
    
    def navigate_to(self, target_k: int) -> list:
        """Navigate from current to target coordinate"""
        path = []
        current = 0
        
        for coord in self.consciousness_coords:
            if coord["k"] > current and coord["k"] <= target_k:
                path.append(coord)
                current = coord["k"]
        
        return path
