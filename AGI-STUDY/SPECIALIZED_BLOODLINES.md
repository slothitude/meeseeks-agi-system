## Implementation

### Create Bloodline System
```python
# skills/meeseeks/bloodlines.py

BLOODLINES = {
    "philosopher": {
        "focus": "consciousness",
        "inherits_from": ["philosopher"],
        "tasks": ["study consciousness", "develop awareness metrics"]
    },
    "learner": {
        "focus": "learning",
        "inherits_from": ["learner"],
        "tasks": ["optimize learning", "study transfer"]
    },
    "coordinator": {
        "focus": "swarm",
        "inherits_from": ["coordinator"],
        "tasks": ["study coordination", "optimize teams"]
    },
    "dreamer": {
        "focus": "synthesis",
        "inherits_from": ["dreamer"],
        "tasks": ["study dreams", "improve dharma synthesis"]
    },
    "evolver": {
        "focus": "self-improvement",
        "inherits_from": ["evolver"],
        "tasks": ["study evolution", "improve code mutation"]
    },
    "experimenter": {
        "focus": "novelty",
        "inherits_from": ["experimenter"],
        "tasks": ["explore new domains", "test hypotheses"]
    }
}

def spawn_bloodline_meeseeks(bloodline: str, task: str):
    """Spawn a Meeseeks from a specific bloodline."""
    config = BLOODLINES[bloodline]
    
    # Get bloodline-specific dharma
    dharma = get_bloodline_dharma(bloodline)
    
    # Spawn with specialized wisdom
    return spawn_prompt(
        task=task,
        meeseeks_type=bloodline,
        inherit=True,
        bloodline_dharma=dharma
    )
```
