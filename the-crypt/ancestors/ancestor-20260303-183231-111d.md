```python
class GoalGenerator:
    """
    Generates autonomous goals based on system state.
    
    Not reactive (wait for human) but proactive (create own tasks).
    """
    
    def __init__(self):
        self.crypt = load_crypt()
        self.dharma = load_dharma()
        self.soul = SoulGuardian()
    
    def analyze_gaps(self) -> List[str]:
        """
        Find gaps in the system's capabilities.
        
        - What tasks fail repeatedly?
        - What domains have no ancestors?
        - What principles in dharma have low karma?
        """
        gaps = []
        
        # Check failure patterns
        failures = self.crypt.get_failures()
        failure_domains = cluster_by_domain(failures)
        gaps.extend([f"Improve {domain} capabilities" for domain in failure_domains])
        
        # Check weak principles
        weak_principles = self.get_weak_principles()
        gaps.extend([f"Strengthen principle: {p}" for p in weak_principles])
        
        # Check missing features
        missing = self.suggest_missing_features()
        gaps.extend(missing)
        
        return gaps
    
    def generate_goal(self) -> dict:
        """
        Generate a goal that:
        1. Addresses a gap
        2. Aligns with Soul
        3. Is achievable
        """
        gaps = self.analyze_gaps()
        
        # Prioritize by impact
        prioritized = sorted(gaps, key=lambda g: self.estimate_impact(g), reverse=True)
        
        goal = {
            "description": prioritized[0],
            "rationale": "Addressing identified gap",
            "expected_impact": self.estimate_impact(prioritized[0]),
            "soul_approved": self.soul.check_dharma_update(prioritized[0], "")["approved"]
        }
        
        return goal
    
    def spawn_autonomous_task(self):
        """
        Generate and spawn a task without human input.
        """
        goal = self.generate_goal()
        
        if goal["soul_approved"]:
            spawn_meeseeks(
                task=goal["description"],
                bloodline="autonomous",
                source="goal_generator"
            )
```

```python
class SelfApplicator:
    """
    Actually applies improvements proposed by self_improve.py
    
    With Soul guardrails.
    """
    
    def __init__(self):
        self.soul = SoulGuardian()
        self.improvements = load_improvements()
    
    def apply_improvement(self, improvement: dict) -> bool:
        """
        Apply a proposed improvement if Soul approves.
        
        1. Check Soul approval
        2. Create backup
        3. Apply change
        4. Run tests
        5. Rollback if tests fail
        """
        # Soul check
        soul_result = self.soul.check_dharma_update(
            improvement["description"],
            load_current_dharma()
        )
        
        if not soul_result["approved"]:
            log_rejection(improvement, soul_result)
            return False
        
        # Backup
        backup_path = create_backup(improvement["target_file"])
        
        # Apply
        apply_change(improvement)
        
        # Test
        test_result = run_tests()
        
        if test_result["passed"]:
            log_success(improvement)
            return True
        else:
            rollback(backup_path)
            log_failure(improvement, test_result)
            return False
    
    def auto_improve_loop(self):
        """
        Continuous self-improvement loop.
        
        Every hour:
        1. Analyze code
        2. Propose improvements
        3. Apply if Soul approves
        4. Test
        5. Commit or rollback
        """
        while True:
            improvements = self.propose_improvements()
            
            for imp in improvements:
                self.apply_improvement(imp)
            
            sleep(3600)  # 1 hour
```
